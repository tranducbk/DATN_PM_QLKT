% --- Slide 6: Chuỗi danh hiệu hằng năm — Logic cốt lõi ⭐ ---
\begin{frame}{Chuỗi danh hiệu hằng năm --- Logic cốt lõi}
  \begin{columns}[T,onlytextwidth]
    \column{0.42\textwidth}
    \textbf{Chuỗi cá nhân}
    \begin{block}{}
      \scriptsize
      \texttt{CSTĐCS × 2 năm}\\
      \quad$\downarrow$\\
      \texttt{BKBQP (cycle 2y)}\\
      \quad$\downarrow$ 3y + 1 BKBQP/3y\\
      \texttt{CSTĐTQ (cycle 3y)}\\
      \quad$\downarrow$ 7y + 3 BKBQP\\
      \quad\quad + 2 CSTĐTQ + NCKH/năm\\
      \texttt{BKTTCP \quad lifetime}
    \end{block}

    \textbf{Chuỗi đơn vị}
    \begin{block}{}
      \scriptsize
      \texttt{ĐVQT × 2y $\to$ BKBQP $\to$\\7y + 3 BKBQP/7y $\to$ BKTTCP}\\
      (đơn vị có thể nhận lặp lại)
    \end{block}

    \column{0.55\textwidth}
    \textbf{Quy tắc đặc biệt}\\[0.2cm]

    \begin{block}{Cửa sổ trượt 3y / 7y}
      \footnotesize BKBQP của chu kỳ trước rơi ra $\to$ bắt buộc có BKBQP mới ở chu kỳ hiện tại
    \end{block}

    \begin{block}{Lifetime BKTTCP cá nhân}
      \footnotesize Chỉ nhận một lần. Hệ thống chặn các đề xuất sau với thông báo cụ thể, do quy định chưa có danh hiệu cao hơn
    \end{block}

    \begin{block}{Lỡ đợt}
      \footnotesize Chu kỳ vẫn tiếp tục đếm. Chu kỳ kế lại được xét, không yêu cầu đứt chuỗi CSTĐCS / ĐVQT
    \end{block}
  \end{columns}
\end{frame}

    \end{block}

    \scriptsize\textbf{4 kịch bản nổi bật:}
    \begin{enumerate}\scriptsize
      \item \textbf{Race condition} --- 2 admin duyệt song song, chỉ 1 thắng
      \item \textbf{BKTTCP eligibility} --- cửa sổ trượt 7y + lifetime
      \item \textbf{Tampering} --- server vẫn ghi đúng dù admin sửa payload
      \item \textbf{Vòng đời thực tế} --- quân nhân lên cấp danh hiệu
    \end{enumerate}

    \column{0.4\textwidth}
    \textbf{Triển khai mạng nội bộ}\\[0.2cm]
    \scriptsize
    \begin{enumerate}
      \item Cài \texttt{node\_modules} trên máy có Internet, chuyển sang máy nội bộ
      \item Tạo CSDL PostgreSQL rỗng + cấu hình \texttt{.env}
      \item \texttt{npm run setup} --- tạo bảng và SUPER\_ADMIN (idempotent)
      \item \texttt{npm run serve} --- build + khởi chạy bằng PM2
    \end{enumerate}

    \begin{block}{PM2 production}
      \scriptsize Auto-restart, RAM 500MB, log rotation. Hệ thống vận hành \textbf{hoàn toàn offline} sau khi cài đặt.
    \end{block}
